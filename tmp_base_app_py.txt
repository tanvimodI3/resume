_cohere = ChatCohere(
    model="command-r-plus-08-2024",
    cohere_api_key=os.getenv("COHERE_API_KEY"),
    temperature=0.1,
)

def _ask_cohere(prompt: str) -> str:
    """Sends a prompt to Cohere and returns the text response."""
    response = _cohere.invoke([HumanMessage(content=prompt)])
    # Clean up standard json markdown formatting
    text = response.content.strip()
    if text.startswith("```json"):
        text = text[len("```json"):].strip()
    if text.endswith("```"):
        text = text[:-3].strip()
    return text

# ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ

def extract_text_from_file(file_path: str) -> str:
    """Loads a PDF, DOCX, or TXT and returns the full concatenated text."""
    ext = file_path.lower().split('.')[-1]
    
    if ext == 'pdf':
        loader = PyMuPDFLoader(file_path)
    elif ext == 'docx':
        loader = Docx2txtLoader(file_path)
    elif ext == 'txt':
        loader = TextLoader(file_path, encoding='utf-8')
    else:
        raise ValueError(f"Unsupported file extension: {ext}")
        
    docs = loader.load()
    return "\n".join(doc.page_content for doc in docs)

def extract_info_with_cohere(resume_text: str) -> dict:
    """
    Uses Cohere to extract structured candidate info from raw resume text.
    Returns: name, email, phone, experience, profiles
    """
    prompt = f"""
You are an expert resume parser. Extract the following fields from the resume below.

Return ONLY a valid JSON object. No explanation, no markdown, no code fences. Just raw JSON.

Fields to extract:
- "name": Full name of the candidate (string)
- "email": Email address (string or null)
- "phone": Phone number (string or null)
- "experience": Total years of experience as a short string like "2 years" or "Fresher" (string)
- "profiles": List of profile URLs (GitHub, LinkedIn, Portfolio, etc.) found in the resume (list of strings)

RESUME TEXT:
{resume_text[:4000]}
"""
    raw = _ask_cohere(prompt)
    try:
        json_match = re.search(r"\{.*\}", raw, re.DOTALL)
        if json_match:
            return json.loads(json_match.group(0))
    except (json.JSONDecodeError, AttributeError):
        pass

    return {"name": "Not found", "email": None, "phone": None, "experience": "Unknown", "profiles": []}

def normalize_skills_with_cohere(skills: list[str]) -> list[dict]:
    """
    Uses Cohere to normalize skill names and group them into categories.
    Returns a list of {skill, category} dicts.
    """
    if not skills:
        return []

    prompt = f"""
You are a skill taxonomy expert. Normalize the following skill list and categorize each skill.

Rules:
- Fix abbreviations and synonyms (e.g. "JS" ΓåÆ "JavaScript", "K8s" ΓåÆ "Kubernetes", "Mongo" ΓåÆ "MongoDB")
- Assign each skill to exactly one of these categories:
  "Languages", "Frontend", "Backend", "Databases", "DevOps & Cloud", "AI & ML",
  "Testing", "Tools & Version Control", "Soft Skills", "Other"

Return ONLY a JSON array. No explanation. No markdown. No code fences. Just raw JSON.

Format:
[{{"skill": "JavaScript", "category": "Languages"}}, ...]

"""
    raw = _ask_cohere(prompt)
    try:
        arr_match = re.search(r"\[.*\]", raw, re.DOTALL)
        if arr_match:
            return json.loads(arr_match.group(0))
    except (json.JSONDecodeError, AttributeError):
        pass

    # Fallback: return unnormalized with unknown category
    return [{"skill": s, "category": "Other"} for s in skills]

def chunk_text(text: str) -> list[str]:
    """Splits raw text into chunks using LangChain's RecursiveCharacterTextSplitter."""
    splitter = RecursiveCharacterTextSplitter(
        chunk_size=500,
        chunk_overlap=50,
        separators=["\n\n", "\n", " ", ""],
    )
    return splitter.split_text(text)

def build_vectorstore(chunks: list[str], embedding_model) -> Chroma:
    """Stores resume chunks in an in-memory ChromaDB collection."""
    from langchain_core.documents import Document
    docs = [Document(page_content=c) for c in chunks]
    return Chroma.from_documents(documents=docs, embedding=embedding_model)

# STEP 8: EMBEDDING SIMILARITY SCORE
# ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ

def compute_embedding_score(vectorstore: Chroma, jd_text: str, k: int = 5) -> float:
    """
    Searches the resume vectorstore with the full JD text.
    Converts ChromaDB distance scores ΓåÆ match percentage.
    """
    results = vectorstore.similarity_search_with_score(jd_text, k=k)
    if not results:
        return 0.0
    total = sum(max(0.0, (1.0 - (score / 2.0)) * 100) for _, score in results)
    return round(total / len(results), 2)

def evaluate_with_cohere(resume_text: str, jd_text: str) -> dict:
    """
    Uses Cohere to qualitatively evaluate the resume against the JD.
    Returns: match_percentage (int), missing_skills (list), strengths (list)
    """
    prompt = f"""
You are a senior ATS (Applicant Tracking System) evaluator with 10+ years of hiring experience.

Evaluate how well the candidate's resume matches the job description below.

JOB DESCRIPTION:
{jd_text.strip()}

RESUME:
{resume_text[:3500]}

Respond ONLY with a valid JSON object. No explanation. No markdown. No code fences.

Format:
{{
  "match_percentage": <integer 0-100>,
  "missing_skills": ["skill1", "skill2"],
  "strengths": ["strength1", "strength2"]
}}
"""
    raw = _ask_cohere(prompt)
    try:
        json_match = re.search(r"\{.*\}", raw, re.DOTALL)
        if json_match:
            return json.loads(json_match.group(0))
    except (json.JSONDecodeError, AttributeError):
        pass
    return {"match_percentage": 0, "missing_skills": [], "strengths": []}

# STEP 10: FINAL SCORE
# ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ

def compute_final_score(embedding_score: float, llm_score: float) -> float:
    """Final Score = 0.7 ├ù embedding_score + 0.3 ├ù llm_score"""
    return round(0.7 * embedding_score + 0.3 * llm_score, 2)

# DISPLAY ΓÇö Clean Terminal Output
# ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ

def display_results(info: dict, normalized_skills: list[dict], embedding_score: float,
                    llm_score: float, final_score: float, llm_eval: dict):
    """Prints the final ATS report. No evidence chunks."""

    W = 54  # box width

    def line(text="", pad=True):
        if pad:
            print(f"  {text}")
        else:
            print(text)

    print()
    print("=" * (W + 4))
    print("| " + "ATS CANDIDATE EVALUATION REPORT".center(W) + " |")
    print("=" * (W + 4))

    # ΓöÇΓöÇ Candidate Info ΓöÇΓöÇ
    print()
    line(f"Name        : {info.get('name', 'Not found')}")
    line(f"Email       : {info.get('email') or 'Not found'}")
    line(f"Phone       : {info.get('phone') or 'Not found'}")
    line(f"Experience  : {info.get('experience') or 'Not found'}")

    profiles = info.get("profiles", [])
    if profiles:
        line(f"Profiles    : {' | '.join(profiles)}")
    else:
        line("Profiles    : Not found")

    # ΓöÇΓöÇ Skills ΓöÇΓöÇ
    print()
    # Group by category
    by_cat: dict[str, list[str]] = {}
    for s in normalized_skills:
        by_cat.setdefault(s.get("category", "Other"), []).append(s.get("skill", ""))

    total_skills = sum(len(v) for v in by_cat.values())
    line(f"Skills ({total_skills} found):")
    for cat, skills in by_cat.items():
        label = f"[{cat}]"
        line(f"  {label:<26} {', '.join(skills)}")

    # ΓöÇΓöÇ Scores ΓöÇΓöÇ
    print()
    print("  " + "-" * (W - 2))
    line(f"Embedding Score  :  {embedding_score:.2f}%")
    line(f"LLM Score        :  {llm_score:.2f}%")
    print()
    line(f"-> FINAL MATCH SCORE :  {final_score:.2f}%")
    print("  " + "-" * (W - 2))

    # ΓöÇΓöÇ Missing Skills ΓöÇΓöÇ
    missing = llm_eval.get("missing_skills", [])
    if missing:
        print()
        line(f"Missing Skills  : {', '.join(missing)}")

    # ΓöÇΓöÇ Strengths ΓöÇΓöÇ
    strengths = llm_eval.get("strengths", [])
    if strengths:
        print()
        line("Strengths:")
        for s in strengths:
            line(f"  ΓÇó {s}")

    print()
    print("=" * (W + 4))
    print()

# MAIN PIPELINE
# ΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇΓöÇ

def run_pipeline(file_path: str, job_description: str):
    print(f"\n  Processing: {file_path}")
    print(f"  JD        : {job_description.strip()[:80]}...\n")

    # Step 1 ΓÇö Extract text
    print("  [1/8] Extracting text from file...")
    resume_text = extract_text_from_file(file_path)

    # Step 2 ΓÇö Extract candidate info
    print("  [2/8] Extracting candidate info (Cohere)...")
    info = extract_info_with_cohere(resume_text)

    # Step 3 ΓÇö Extract skills
    print("  [3/8] Extracting skills (Cohere)...")
    raw_skills = extract_skills_with_cohere(resume_text)

    # Step 4 ΓÇö Normalize skills
    print("  [4/8] Normalizing skills (Cohere)...")
    normalized_skills = normalize_skills_with_cohere(raw_skills)

    # Step 5 ΓÇö Chunk resume text
    print("  [5/8] Chunking resume text...")
    chunks = chunk_text(resume_text)

    # Step 6 + 7 ΓÇö Embed + store in ChromaDB
    print("  [6/8] Generating embeddings + storing in ChromaDB...")
    embedding_model = CohereEmbeddings(
        model="embed-english-v3.0",
        cohere_api_key=os.getenv("COHERE_API_KEY")
    )
    vectorstore = build_vectorstore(chunks, embedding_model)

    # Step 8 ΓÇö Embedding similarity score
    print("  [7/8] Computing embedding similarity score...")
    k = min(5, len(chunks))
    embedding_score = compute_embedding_score(vectorstore, job_description, k=k)

    # Step 9 ΓÇö Cohere LLM evaluation
    print("  [8/8] Running Cohere final evaluation...")
    llm_eval = evaluate_with_cohere(resume_text, job_description)
    llm_score = float(llm_eval.get("match_percentage", 0))

    # Step 10 ΓÇö Final score
    final_score = compute_final_score(embedding_score, llm_score)

    # Display
    display_results(info, normalized_skills, embedding_score, llm_score, final_score, llm_eval)

    # Step 11 ΓÇö Save to Database
    print("  [9/9] Saving candidate info to database...")
    from backend.services.db_service import save_candidate_data
    save_candidate_data(
        name=info.get("name"),
        email=info.get("email"),
        score=final_score,
        pdf_url=file_path
    )

if __name__ == "__main__":
    # Accept optional CLI argument: python app.py resume3.pdf
    resume_file = sys.argv[1] if len(sys.argv) > 1 else "resume3.pdf"

    job_description = """
    MERN stack developer with experience in React, Node.js, Express.js, MongoDB, and REST APIs.
    Strong knowledge of JavaScript and TypeScript. Experience with Docker, CI/CD pipelines,
    and cloud deployment (AWS or GCP). Familiarity with Agile methodology and Git-based workflows.
    """
